            def linearSearch(list1, n, key):
    for i in range(0, n):
        if (list1[i] == key):
            return i
    return -1

list1 = [1 ,3, 5, 4, 7, 9]
key = int(input("Enter the element to be found : "))
print(key)

n = len(list1)
res = linearSearch(list1, n, key)
if(res == -1):
    print"Element not found"
else:
    print"Element found at index :",res
