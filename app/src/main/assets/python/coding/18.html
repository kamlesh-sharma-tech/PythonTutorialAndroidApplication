18) Program to Reverse a String

            def reverse_string(str):
    str1 = ""
    for i in str:
        str1 = i + str1
    return str1

str = input("Enter a string : ")
print(str)
print"\nThe original string is: ",str
print"\nThe reverse string is",reverse_string(str)
