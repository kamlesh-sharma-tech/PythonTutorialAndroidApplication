Python Digital Clock

from tkinter import *
import time

root=Tk()
root.title("Python Digital Clock")
root.geometry("1250x700+150+50")
root.resizable(False,False)
root.configure(bg="#ff6699")

def clockTime():
    h=str(time.strftime("%H"))
    m=str(time.strftime("%M"))
    s=str(time.strftime("%S"))
    label_hour_digit.configure(text=h)
    label_minute_digit.configure(text=m)
    label_second_digit.configure(text=s)

    if int(h)>12 and int(m)>0:
        label_check_am_pm.config(text="PM")

    label_hour_digit.after(200,clockTime)

def exitWindow():
    root.destroy()

title=Label(root,text="DIGITAL CLOCK USING PYTHON",font=("arial",25,"bold"),fg="darkslategray",bg="lightgray",underline=0)
title.pack(pady=(80,0))

label_hour_digit=Label(root,text="",width=10,bd=5,relief=RAISED,height=5,font=("times new roman",28,"bold"),bg="orange")
label_hour_digit.place(x=150,y=200)

label_text_hour=Label(root,text="HOUR",width=20,bd=5,relief=RAISED,height=3,font=("",15),bg="orange")
label_text_hour.place(x=150,y=430)

label_minute_digit=Label(root,text="",width=10,bd=5,relief=RAISED,height=5,font=("times new roman",28,"bold"),bg="#0099cc")
label_minute_digit.place(x=390,y=200)

label_text_hour=Label(root,text="MINUTE",width=20,bd=5,relief=RAISED,height=3,font=("",15),bg="#0099cc")
label_text_hour.place(x=390,y=430)

label_second_digit=Label(root,text="",width=10,bd=5,relief=RAISED,height=5,font=("times new roman",28,"bold"),bg="#6699ff")
label_second_digit.place(x=630,y=200)

label_text_hour=Label(root,text="SECOND",width=20,bd=5,relief=RAISED,height=3,font=("",15),bg="#6699ff")
label_text_hour.place(x=630,y=430)

label_check_am_pm=Label(root,text="PM",width=10,bd=5,relief=RAISED,height=5,font=("times new roman",28,"bold"),bg="#00ff99")
label_check_am_pm.place(x=870,y=200)

label_check_am_pm_text=Label(root,text="NOON",width=20,bd=5,relief=RAISED,height=3,font=("",15),bg="#00ff99")
label_check_am_pm_text.place(x=870,y=430)

clockTime()

exit_btn=Button(root,text="Exit",width=18,height=2,background="darkslategray",fg="white",font=("Arial",12),command=lambda:exitWindow())
exit_btn.pack(side=BOTTOM,pady=30)

root.mainloop()
